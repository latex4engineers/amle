Wenn $n = 2$:
\begin{subequations}
	\begin{equation}
	\binom{2}{0} = \frac{2!}{0!(2-0)!} = \frac{1 \cdot 2}{1 \cdot 2} = 1
	\end{equation}
	\begin{equation}
	\binom{2}{1} = \frac{2!}{1!(2-1)!} =  \frac{1 \cdot 2}{1 \cdot 1} = 2
	\end{equation}
	\begin{equation}
	\binom{2}{2} = \frac{2!}{2!(2-2)!} =  \frac{1 \cdot 2}{2 \cdot 1} = 1
	\end{equation}
\end{subequations}

\begin{align}
(a+b)^2 = \sum_{i=0}^2 {2 \choose i} a^{2-i} \cdot b^i &= \binom{2}{0} a^2 \cdot b^0 + \binom{2}{1} a^{2-1} \cdot b^1 + \binom{2}{2} a^{2-2} \cdot b^2 \\
&= a^2 + 2 \cdot a \cdot b + b^2 \notag 
\end{align}

\begin{gnuplot}[terminal=pdf]

set grid ytics lt 1 lw 0.5 lc rgb '#bbbbbb'

show key

plot 'wegzeit.csv' using 1:2 with linespoints lt 1 lc rgb '#000000' pt 9 lw 2 title 's_1',\
'wegzeit.csv' using 1:3 with linespoints lt 0 lc rgb '#000000' pt 2 lw 3 title 's_2'
